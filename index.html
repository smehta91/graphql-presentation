Type System



type Squad {
  name
}

          

type Match {
  id: Int!
  invite: {
    count: Int
  }
  squads: [Squad]
}
          

            type Game {
              id: Int!
              name: String!
              matches: [Match]
            }
          

            type Query {
              game(name: String!): Game
              games: [Game]
            }

Every field Is a Function


            query {
              game(name: "cricket") {
                  id
                  matches {
                      id
                      squads {
                          name
                      }
                      invites {
                          count
                      }
                  }
              }
          }

Mutations


mutation ChangeStaff {
  changeName(productId: 123, newName: "Cheesecake") {
    id, version
  }

  setDescription(productId: 123, description: "Delicious!") {
    id, version
  }
}

subscription ProductEvents {
  nameChanged(productId: 123) { name }
  productDeleted { id }
}
